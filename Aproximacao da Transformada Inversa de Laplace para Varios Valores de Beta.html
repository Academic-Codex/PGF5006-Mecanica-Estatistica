import numpy as np
import matplotlib
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

%matplotlib widget

# Parmetros para a aproximao da funo delta via exponencial
x = np.linspace(-5, 5, 1000)  # Eixo x
beta_vals = np.linspace(0.1, 2, 30)  # Vrios valores de beta

# Preparar o grid para o grfico 3D
X, B = np.meshgrid(x, beta_vals)

# Funo exponencial para a transformada de Laplace inversa
delta_approx_3D = np.exp(-B * np.abs(X))

# Criar o grfico 3D interativo
fig = plt.figure(figsize=(10, 7))
ax = fig.add_subplot(111, projection='3d')

# Plotar a superfcie
surf = ax.plot_surface(X, B, delta_approx_3D, cmap='viridis')

# Adicionar rtulos e ttulo
ax.set_title("Aproximao da Transformada Inversa de Laplace para Vrios Valores de Beta")
ax.set_xlabel('x')
ax.set_ylabel('Beta')
ax.set_zlabel('Intensidade')

# Adicionar barra de cores para representar a intensidade
fig.colorbar(surf, ax=ax, shrink=0.5, aspect=5)

# Garantir que o backend seja interativo
plt.ion()

# Mostrar o grfico com interatividade
plt.show()
